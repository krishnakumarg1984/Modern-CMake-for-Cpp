cmake_minimum_required(VERSION 3.20.0)
project(Scope)

function(Inner)
    message("  > Inner: ${V}")
    set(V 3 PARENT_SCOPE)
    message("  < Inner: ${V}")
endfunction()

function(Outer)
    message(" > Outer: ${V}")
    set(V 2)
    inner()
    message(" < Outer: ${V}")
endfunction()

set(V 1)
message("> Global: ${V}")
outer()
message("< Global: ${V}")
